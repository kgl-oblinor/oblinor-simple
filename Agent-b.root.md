import { useState, useEffect, useRef, useMemo, useCallback } from 'react';
import { THEME_ENHANCED } from '../constants/theme';

// PERFORMANCE-CRITICAL: Debounced resize with RAF optimization
const createOptimizedResize = (callback: Function, delay = 16) => {
  let timeoutId: NodeJS.Timeout;
  let rafId: number;
  
  const optimized = (...args: any[]) => {
    clearTimeout(timeoutId);
    if (rafId) cancelAnimationFrame(rafId);
    
    timeoutId = setTimeout(() => {
      rafId = requestAnimationFrame(() => callback.apply(null, args));
    }, delay);
  };
  
  optimized.cancel = () => {
    clearTimeout(timeoutId);
    if (rafId) cancelAnimationFrame(rafId);
  };
  
  return optimized;
};

// INTELLIGENT CACHING LAYER (inspired by DesignArkitekt Pro)
let breakpointCache = {
  width: 0,
  isMobile: false,
  isTablet: false,
  isDesktop: false,
  timestamp: 0
};

export const useResponsiveHarmonic = () => {
  const [viewport, setViewport] = useState(() => {
    const width = typeof window !== 'undefined' ? window.innerWidth : 1024;
    return {
      width,
      height: typeof window !== 'undefined' ? window.innerHeight : 768,
      timestamp: Date.now()
    };
  });

  const optimizedResizeRef = useRef<any>();

  useEffect(() => {
    // HYBRID: Advanced debouncing + intelligent caching
    const handleResize = () => {
      const newWidth = window.innerWidth;
      const newHeight = window.innerHeight;
      const now = Date.now();
      
      // CACHE OPTIMIZATION: Skip unnecessary updates
      if (Math.abs(newWidth - viewport.width) < 5 && 
          now - viewport.timestamp < 50) {
        return; // Skip micro-adjustments within 50ms
      }
      
      setViewport({ width: newWidth, height: newHeight, timestamp: now });
      
      // Update global cache for utility functions
      breakpointCache = {
        width: newWidth,
        isMobile: newWidth <= THEME_ENHANCED.breakpoints.mobile,
        isTablet: newWidth > THEME_ENHANCED.breakpoints.mobile && newWidth <= 1024,
        isDesktop: newWidth > 1024,
        timestamp: now
      };
    };

    optimizedResizeRef.current = createOptimizedResize(handleResize, 16);
    window.addEventListener('resize', optimizedResizeRef.current, { passive: true });
    
    // Initialize cache
    handleResize();

    return () => {
      window.removeEventListener('resize', optimizedResizeRef.current);
      optimizedResizeRef.current?.cancel();
    };
  }, []);

  return useMemo(() => {
    const isMobile = viewport.width <= THEME_ENHANCED.breakpoints.mobile;
    const isTablet = viewport.width > THEME_ENHANCED.breakpoints.mobile && viewport.width <= 1024;
    const isDesktop = viewport.width > 1024;

    return {
      // EXACT API AS REQUESTED
      isMobile,
      isTablet,
      isDesktop,

      height: viewport.height,

        if (isMobile) return mobile;
        if (isTablet) return tablet;
        return desktop;
      }, [isMobile, isTablet]),

      getValue: useCallback(<T>(values: { mobile: T; tablet?: T; desktop: T }): T => {
        if (isMobile) return values.mobile;
        if (isTablet && values.tablet) return values.tablet;
        return values.desktop;
      }, [isMobile, isTablet]),
      
      // PERFORMANCE-OPTIMIZED SPACING
      spacing: useCallback((size: 'xs' | 'sm' | 'md' | 'lg' | 'xl' = 'md') => {
        return THEME_ENHANCED.spacing.grid[size][isMobile ? 'mobile' : 'desktop'];
      }, [isMobile]),

        const scale = THEME_ENHANCED.typography.scale[variant];
        if (isMobile) return scale.mobile;

  const currentWidth = window.innerWidth;
  const now = Date.now();
  
  // Cache hit optimization  
  if (responsiveCache && 
      now - responsiveCache.timestamp < CACHE_DURATION &&
      Math.abs(currentWidth - responsiveCache.width) < 20) {

  // Calculate fresh state

    isMobile: currentWidth <= RESPONSIVE.mobile,
    isTablet: currentWidth > RESPONSIVE.mobile && currentWidth <= RESPONSIVE.tablet, 
    isDesktop: currentWidth > RESPONSIVE.tablet
  };
  
  // Update cache
  responsiveCache = { width: currentWidth, state, timestamp: now };
  return state;
};

export const getAlphaColor = (
  color: string, 
  alpha: keyof typeof THEME_EXTENDED.alpha
): string => {
  const alphaValue = THEME_EXTENDED.alpha[alpha];
  // Smart color parsing - handles both hex and rgb
  if (color.startsWith('#')) {
    // Convert hex to rgba
    const r = parseInt(color.slice(1, 3), 16);
    const g = parseInt(color.slice(3, 5), 16); 
    const b = parseInt(color.slice(5, 7), 16);
    return `rgba(${r}, ${g}, ${b}, ${alphaValue})`;
  }
  // If already rgba/rgb, extract and modify alpha
  return color.replace(/rgba?\(([^)]+)\)/, (match, values) => {
    const parts = values.split(',').slice(0, 3);
    return `rgba(${parts.join(', ')}, ${alphaValue})`;
  });
};

// Memoized calculation to avoid repeated window.innerWidth calls
let cachedBreakpoint: { width: number; isMobile: boolean; timestamp: number } | null = null;
const CACHE_DURATION = 100; // 100ms cache prevents excessive calculations

export const isMobileCached = (): boolean => {
  const now = Date.now();
  const currentWidth = window.innerWidth;
  
  if (
    !cachedBreakpoint ||
    now - cachedBreakpoint.timestamp > CACHE_DURATION ||
    Math.abs(currentWidth - cachedBreakpoint.width) > 10 // 10px threshold
  ) {
    cachedBreakpoint = {
      width: currentWidth,
      isMobile: currentWidth <= THEME_EXTENDED.breakpoints.mobile,
      timestamp: now
    };
  }
  
  return cachedBreakpoint.isMobile;
};
